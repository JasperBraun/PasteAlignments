316 template <class ParameterType>
317 ParameterType ArgumentMap::GetValue(const std::string& name, int pos) {
318  int id{parameters_.GetId(name)};
319  std::function<ParameterType(const std::string&)> converter{
320  parameters_.ConversionFunction<ParameterType>(name)};
321  std::stringstream error_message;
322 
323  // Test if argument at position `pos` was assigned.
324  if (arguments_.at(id).size() <= pos) {
325  error_message << "Attempted to access argument at position '" << pos
326  << "' for parameter named '" << name << "' but only '"
327  << arguments_.at(id).size()
328  << "' arguments were assigned.";
329  throw exceptions::ValueAccessError(error_message.str());
330  }
331  // Test if parameter has a conversion function.
332  if (converter == nullptr) {
333  error_message << "Parameter identified by '" << name << "' has no"
334  << " conversion function associated with it.";
335  throw exceptions::ValueAccessError(error_message.str());
336  }
337  // Test if `name` is a flag.
338  if (parameters_.flags().count(id)) {
339  error_message << "Attempted to use `ArgumentMap::GetValue` to check if flag"
340  " named: '" << name << "' was set. Use"
341  " `ArgumentMap::IsSet` to test flag values.";
342  throw exceptions::ValueAccessError(error_message.str());
343  }
344 
345  // Compute value only if it wasn't computed before.
346  if (value_lists_.at(id).size() <= pos
347  || !value_lists_.at(id).at(pos).has_value()) {
348  value_lists_.at(id).reserve(pos);
349  for (int i = value_lists_.at(id).size(); i < pos; ++i) {
350  value_lists_.at(id).emplace_back();
351  }
352  value_lists_.at(id).emplace_back(converter(arguments_.at(id).at(pos)));
353  }
354  return std::any_cast<ParameterType>(value_lists_.at(id).at(pos));
355 }
