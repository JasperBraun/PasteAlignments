316 template <class ParameterType>
317 std::function<ParameterType(const std::string&)>
318 ParameterMap::ConversionFunction(const std::string& name) const {
319  std::stringstream error_message;
320  try {
321  // Note placement of * and & here to obtain const reference.
322  return std::any_cast<std::function<ParameterType(const std::string&)>>(
323  converters_.at(GetId(name)));
324  } catch (const std::out_of_range& e) {
325  error_message << "Expected the name of a parameter contained in the"
326  << " object. No parameter with name: '" << name << "' was"
327  << " found.";
328  throw exceptions::ParameterAccessError(error_message.str());
329  } catch (const std::bad_any_cast& e) {
330  error_message << "Expected the argument to match the original `Parameter`"
331  << " object's template argument. Function was called with:"
332  << " '" << typeid(ParameterType).name() << "', but original"
333  << " `Parameter` object has conversion function type"
334  << converters_.at(GetId(name)).type().name() << "'.";
335  throw exceptions::ParameterAccessError(error_message.str());
336  }

341 template<class ParameterType>
342 ParameterMap& ParameterMap::operator()(Parameter<ParameterType> parameter) {
343  int id{static_cast<int>(parameter_configurations_.size())};
344  std::vector<std::string> names{parameter.configuration().names()};
345  std::any converter{
346  std::make_any<std::function<ParameterType(const std::string&)>>(
347  std::move(parameter.converter()))};
348  ParameterCategory parameter_category{parameter.configuration().category()};
349  int parameter_position{parameter.configuration().position()};
350  std::stringstream error_message;
351  int other_id;
352 
353  // Preconditions.
354  if (names.empty()) {
355  error_message << "Parameter must be given at least one name.";
356  throw exceptions::ParameterRegistrationError(error_message.str());
357  }
358  for (const std::string& name : names) {
359  if (name_to_id_.count(name)) {
360  error_message << "Name '" << name << "' already taken by another"
361  << " parameter.";
362  throw exceptions::ParameterRegistrationError(error_message.str());
363  }
364  }
365  if (parameter_category == ParameterCategory::kPositionalParameter
366  && positional_parameters_.count(parameter_position)) {
367 
368  other_id = positional_parameters_.at(parameter_position);
369  error_message << "Position '" << parameter_position << "' for parameter"
370  << " named '" << names.at(0)
371  << "' already taken by parameter named: '"
372  << parameter_configurations_.at(other_id).names().at(0)
373  << "'.";
374  throw exceptions::ParameterRegistrationError(error_message.str());
375  }
376 
377  // Reserve space to trigger exceptions before modifying
378  // Move constructor of std::function isn't 'noexcept' until C++20, swap is.
379  converters_.emplace_back();
380  name_to_id_.reserve(name_to_id_.size() + names.size());
381  parameter_configurations_.reserve(parameter_configurations_.size() + 1);
382  converters_.reserve(converters_.size() + 1);
383  if (parameter.configuration().IsRequired()) {
384  required_parameters_.reserve(required_parameters_.size() + 1);
385  }
386  switch (parameter_category) {
387  case ParameterCategory::kKeywordParameter: {
388  keyword_parameters_.reserve(keyword_parameters_.size() + 1);
389  break;
390  }
391  case ParameterCategory::kFlag: {
392  flags_.reserve(flags_.size() + 1);
393  break;
394  }
395  default: {
396  // Positional parameters are inserted into std::map; no reserve needed.
397  }
398  }
399 
400  // Insert names.
401  for (std::string& name : names) {
402  name_to_id_.emplace(std::move(name), id);
403  }
404  // Insert converter.
405  converters_.back().swap(converter);
406  // Insert into appropriate categories.
407  if (parameter.configuration().IsRequired()) {
408  required_parameters_.emplace(id);
409  }
410  switch (parameter_category) {
411  case ParameterCategory::kPositionalParameter: {
412  positional_parameters_.emplace(parameter_position, id);
413  break;
414  }
415  case ParameterCategory::kKeywordParameter: {
416  keyword_parameters_.emplace(id);
417  break;
418  }
419  case ParameterCategory::kFlag: {
420  flags_.emplace(id);
421  break;
422  }
423  default: {
424  // Already handled above.
425  }
426  }
427  // Insert configuration object.
428  parameter_configurations_.emplace_back(std::move(parameter.configuration()));
429  return *this;
430 }
