208  inline void AddDefault(std::string argument) {
209  std::stringstream error_message;
210  if (argument.empty()) {
211  error_message << "Attempted to add empty default argument; (parameter"
212  << " name: '" << PrimaryName() << "').";
213  throw exceptions::ParameterConfigurationError(error_message.str());
214  }

224  inline void SetDefault(std::vector<std::string> default_arguments) {
225  std::stringstream error_message;
226  for (const std::string& argument : default_arguments) {
227  if (argument.empty()) {
228  error_message << "Attempted to assign list of default arguments"
229  << " containing an empty argument; (parameter name: '"
230  << PrimaryName() << "').";
231  throw exceptions::ParameterConfigurationError(error_message.str());
232  }
233  }
234  default_arguments_ = std::move(default_arguments);
235  }
